        private void aç(object sender, EventArgs e)
        {
            // using ile bir dosya açma diyaloðu oluþturuyoruz

                // eðer diyalogdan baþarýlý sonuç dönerse
                if (dialog.ShowDialog() == DialogResult.OK)
                {
                    // seçilen resmi alýyoruz
                    Bitmap resim = new Bitmap(dialog.FileName);

                    // resmi forma oturtmak için gerekli boyutu hesaplýyoruz
                    Size yeniboyut = resmiortalama(resim.Size, 600, 500);

                    // boyutlandýrýlmýþ resmi oluþturuyoruz

                    // ve kutuya atýyoruz
                    pictureBox1.Image = new Bitmap(yeniresim);

                    // arkaya eklediðimiz hidden box içerisine resmin orjinal boyutunu atýyoruz
                    hidden_box.Visible = false;
                    hidden_box.Width = resim.Width;
                    hidden_box.Height = resim.Height;
                    hidden_box.Image = new Bitmap(resim);
                    

                   // this.realImage = resim;
                    this.sabitresim = resim;
                   // pictureBox1.Visible = true;
                   

                }
            }

        //resmi ortalayýp sabit alan tutmak için
        public static Size resmiortalama(Size src, int maxWidth, int maxHeight)
        {
            decimal rnd = Math.Min(maxWidth / (decimal)src.Width, maxHeight / (decimal)src.Height);
            return new Size((int)Math.Round(src.Width * rnd), (int)Math.Round(src.Height * rnd));
        }

        /*
         * invert iþlemi için resmin her pikselini alýp
         * R, G ve B olan üç deðerini 255'den çýkarýp sonuçlarý yeni resimde toparlýyoruz.
         * */
        private void ýnvertToolStripMenuItem_Click(object sender, EventArgs e)
        {
            if (pictureBox1.Image != null)
            {
                Bitmap real = new Bitmap(hidden_box.Image);
                for (int i = 0; i < real.Width; i++)
                {
                    for (int j = 0; j < real.Height; j++)
                    {
                        Color inv = real.GetPixel(i, j);
                        inv = Color.FromArgb(255, (255 - inv.R), (255 - inv.G), (255 - inv.B));
                        real.SetPixel(i, j, inv);
                    }
                }
                //invert terslenen resmi ayný boyutta açmak için bitmapte alan alýyoruz
                Bitmap boyutlandirilmisresim = new Bitmap(real, resmiortalama(real.Size, 600, 500));

                hidden_box.Image = real;
                pictureBox1.Image = boyutlandirilmisresim;

            }
        }

        /*
         * saða aynalama için önce resmin iki katý geniþlikte yeni resim oluþturuyoruz
         * ve örneðin geniþliði x piksel bir resim varsa:
         * 0. pikseli -> (2*x-0). piksele
         * 1. pikseli -> (2*x-1). piksele
         * taþýyarak saða aynalamayý tamamlýyoruz
         */
        private void saðaToolStripMenuItem_Click(object sender, EventArgs e)
        {
            if (pictureBox1.Image != null)
            {
                Bitmap real = new Bitmap(hidden_box.Image);
                Bitmap aynala = new Bitmap(real.Width * 2, real.Height);
                for (int i = 0; i < real.Width; i++)
                {
                    for (int j = 0; j < real.Height; j++)
                    {
                        Color pixel = real.GetPixel(i, j);
                        aynala.SetPixel(i, j, pixel);
                        aynala.SetPixel((aynala.Width - 1) - i, j, pixel);
                    }
                }

                hidden_box.Width = aynala.Width;
                hidden_box.Height = aynala.Height;
                hidden_box.Image = aynala;

                Bitmap resizedMirrored = new Bitmap(aynala, resmiortalama(aynala.Size, 600, 500));

                pictureBox1.Image = resizedMirrored;
            }
        }

        /*
         * sola aynalama saða aynalamadan farklý gerçekleþiyor.
         * orjinal resim, aynalanmýþ resimde saðda konumlanacaðý için
         * orjinal resimdeki her pikseli saða taþýyoruz ve ayna pikselini oluþturuyoruz.
         * örneðin, geniþliði x piksel bir resim varsa:
         * 0. pikseli -> (x + 0). piksele ve (x - 0). piksele
         * 1. pikseli -> (x + 1). piksele ve (x - 1). piksele
         * taþýyýp sola aynalama iþlemini tamamlýyoruz
         */
        private void solaToolStripMenuItem_Click(object sender, EventArgs e)
        {
            if (pictureBox1.Image != null)
            {
                Bitmap real = new Bitmap(hidden_box.Image);
                Bitmap aynala = new Bitmap(real.Width * 2, real.Height);
                for (int i = 0; i < real.Width; i++)
                {
                    for (int j = 0; j < real.Height; j++)
                    {
                        Color pixel = real.GetPixel(i, j);
                        aynala.SetPixel(real.Width + i, j, pixel);
                        aynala.SetPixel(real.Width - i, j, pixel);
                    }
                }

                hidden_box.Width = aynala.Width;
                hidden_box.Height = aynala.Height;
                hidden_box.Image = aynala;

                Bitmap resizedMirrored = new Bitmap(aynala, resmiortalama(aynala.Size, 600, 500));

                pictureBox1.Image = resizedMirrored;
            }
        }

         * eski resmin (0, 0) pikseli, yeni resimde (0, a-0) konumuna

                        int gray = (int)((pixel.R * 0.3) + (pixel.G * 0.59) + (pixel.B * 0.11));

using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Data;
using System.Drawing;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Windows.Forms;

namespace GoruntuIslemeEditoru
{
    public partial class HistogramForm : Form
    {
        private Image resim;

        public HistogramForm(Image resim)
        {
            this.resim = resim;
            InitializeComponent();
        }

        private void HistogramForm_Load(object sender, EventArgs e)
        {
            this.MaximizeBox = false;
            this.MinimizeBox = false;

            Bitmap resim = new Bitmap(this.resim);

           
            int[] kirmizi = new int[256];
            int[] yesil = new int[256];
            int[] mavi = new int[256];

           
            int maximumKirmiziDeger = 0;
            int maximumYesilDeger = 0;
            int maximumMaviDeger = 0;

            for (int i = 0; i < resim.Width; i++)
            {
                for (int j = 0; j < resim.Height; j++)
                {
                    
                    int KirmiziDeger = resim.GetPixel(i, j).R;
                    int YesilDeger = resim.GetPixel(i, j).G;
                    int MaviDeger = resim.GetPixel(i, j).B;

                   
                    kirmizi[KirmiziDeger]++;
                    yesil[YesilDeger]++;
                    mavi[MaviDeger]++;

                   
                    if (kirmizi[KirmiziDeger] > maximumKirmiziDeger)
                        maximumKirmiziDeger = kirmizi[KirmiziDeger];

                    if (yesil[YesilDeger] > maximumYesilDeger)
                        maximumYesilDeger = yesil[YesilDeger];

                    if (mavi[MaviDeger] > maximumMaviDeger)
                        maximumMaviDeger = mavi[MaviDeger];
                }
            }

            Bitmap histogramKirmizi = new Bitmap(256, 128);
            Bitmap histogramYesil = new Bitmap(256, 128);
            Bitmap histogramMavi = new Bitmap(256, 128);

           
            using (Graphics r = Graphics.FromImage(histogramKirmizi))
            {
                for (int i = 0; i < kirmizi.Length; i++)
                {
                    float height = ((float)kirmizi[i] / (float)maximumKirmiziDeger) * (float)128;
                    r.DrawLine(Pens.Red,
                        new Point(i, histogramKirmizi.Height),
                        new Point(i, (int)(histogramKirmizi.Height - height))
                        );
                }
            }

            using (Graphics g = Graphics.FromImage(histogramYesil))
            {
                for (int i = 0; i < yesil.Length; i++)
                {
                    float height = ((float)yesil[i] / (float)maximumYesilDeger) * (float)128;
                    g.DrawLine(Pens.Green,
                        new Point(i, histogramYesil.Height),
                        new Point(i, (int)(histogramYesil.Height - height))
                        );
                }
            }

            using (Graphics b = Graphics.FromImage(histogramMavi))
            {
                for (int i = 0; i < mavi.Length; i++)
                {
                    float height = ((float)mavi[i] / (float)maximumMaviDeger) * (float)128;
                    b.DrawLine(Pens.Blue,
                        new Point(i, histogramMavi.Height),
                        new Point(i, (int)(histogramMavi.Height - height))
                        );
                }
            }

            
            pictureBox3.Image = histogramKirmizi;
            pictureBox4.Image = histogramYesil;
            pictureBox5.Image = histogramMavi;

        }
    }
